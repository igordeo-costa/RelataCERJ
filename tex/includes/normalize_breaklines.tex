\begin{luacode*}
function normalize_breaklines(input, output)

  local infile = io.open(input,"r")
  if not infile then
    texio.write_nl("Erro: não consegui abrir "..input)
    return
  end

  local outfile = io.open(output,"w")
  if not outfile then
    texio.write_nl("Erro: não consegui criar "..output)
    infile:close()
    return
  end

  local buffer = ""
  local blank_pending = false

  for line in infile:lines() do

    -- Detecta início de novo registro do CSV
    if line:match("^%d%d/%d%d/%d%d%d%d %d%d:%d%d:%d%d,") then

      if buffer ~= "" then
        outfile:write(buffer,"\n")
      end

      buffer = line
      blank_pending = false

    else

      -- Linha em branco
      if line:match("^%s*$") then
        blank_pending = true

      else

        if buffer == "" then
          buffer = line
        else

          if blank_pending then
            buffer = buffer .. "\\\\[1em]" .. line
          else
            buffer = buffer .. "\\\\" .. line
          end

        end

        blank_pending = false
      end

    end

  end

  if buffer ~= "" then
    outfile:write(buffer,"\n")
  end

  infile:close()
  outfile:close()

end
\end{luacode*}
